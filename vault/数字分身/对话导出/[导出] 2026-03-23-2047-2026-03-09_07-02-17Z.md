def _save_messages_md(self, new_messages: List[Dict]):
    """保存到MD文件（每天一个文件，追加写入）"""
    if not new_messages:
        return

    today = datetime.now().strftime("%Y-%m-%d")
    md_file = DIALOG_DIR / f"[对话] {today}.md"

    # 如果是新的一天，创建文件头
    if not md_file.exists():
        header = f"# 对话记录 {today}\n\n"
        md_file.write_text(header, encoding="utf-8")

    # 追加新消息
    with open(md_file, "a", encoding="utf-8") as f:
        for msg in new_messages:
            role = "👤 用户" if msg["role"] == "user" else "🤖 Claude"
            time_str = msg.get("time", "")
            turn = msg.get("turn", 0)
            content = msg.get("content", "")

            f.write(f"\n---\n\n")

            f.write(f"{content}\n")

### rename_files.py (Created)
**Purpose**: Batch rename user input files from .txt to [输入] xxx.md

```python
# -*- coding: utf-8 -*-
"""重命名用户输入文件：.txt → [输入] xxx.md"""

from pathlib import Path

VAULT = Path(__file__).parent.parent
INPUT_DIR = VAULT / "用户输入"

def rename_input_files():
    """把用户输入目录下的.txt文件重命名为[输入] xxx.md"""
    count = 0
    for txt_file in INPUT_DIR.rglob("*.txt"):
        new_name = f"[输入] {txt_file.stem}.md"
        new_path = txt_file.parent / new_name
        txt_file.rename(new_path)
        print(f"OK: {txt_file.name} -> {new_name}")
        count += 1
    print(f"\nTotal: {count} files")
```
