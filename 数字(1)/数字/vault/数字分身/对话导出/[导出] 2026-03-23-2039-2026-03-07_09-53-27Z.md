class LogHandler(FileSystemEventHandler):
    def on_modified(self, event):
        if event.is_directory or not event.src_path.endswith('.md'):
            return

        with open(event.src_path, 'r', encoding='utf-8') as f:

            for line in latest_lines:
                # 1. 检查是否需要触发上帝视角复盘
                if any(keyword in line.lower() for keyword in COMPRESSION_KEYWORDS):
                    self.trigger_alert()
                
                # 2. 清洗数据并写入到干净的数据库或另一个文件中供 UI 读取
                clean_text = self.clean_terminal_symbols(line)

    def clean_terminal_symbols(self, text):
        # 使用正则表达式去掉终端的乱码、ANSI颜色代码、特定的 Bash 符号
