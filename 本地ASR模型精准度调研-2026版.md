## 五、双模型组合策略（最佳方案）

### 策略：FunASR（中文） + Faster-Whisper（多语言）

```python
def smart_transcribe(video_path, language=None):
    """
    智能选择模型转写
    
    Args:
        video_path: 视频路径
        language: 语言（zh/en/None自动检测）
    """
    if language == "zh" or language is None:
        # 先用FunASR尝试（中文最优）
        try:
            from funasr import AutoModel
            model = AutoModel(model="paraformer-zh")
            result = model.generate(input=video_path)
            
            # 检查是否为中文（简单判断）
            text = result[0]["text"]
            chinese_ratio = len([c for c in text if '\u4e00' <= c <= '\u9fff']) / len(text)
            
            if chinese_ratio > 0.3:  # 30%以上中文
                print("使用FunASR（中文优化）")
                return result[0]
        except:
            pass
    
    # 其他语言或FunASR失败，用Faster-Whisper
    print("使用Faster-Whisper（多语言）")
    from faster_whisper import WhisperModel
    model = WhisperModel("large-v3", device="cuda")
    segments, info = model.transcribe(video_path, language=language)
    
    text = "".join([seg.text for seg in segments])
    return {"text": text, "language": info.language}

# 使用
result = smart_transcribe("video.mp4")
print(result["text"])
```
